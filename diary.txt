function ordinal(n){
  const s=["th","st","nd","rd"], v=n%100;
  return n + (s[(v-20)%10] || s[v] || s[0]);
}

function fmtCoverDate(date, format){
  if(format === 'MDY'){
    return date.toLocaleDateString('en-US', {
      month:'long',
      day:'numeric',
      year:'numeric'
    });
  }

  const day = ordinal(date.getDate());
  const monthYear = date.toLocaleDateString('en-AU', {
    month:'long',
    year:'numeric'
  });
  return `${day} ${monthYear}`;
}

function addMonthsKeepingDay(date, count){
  const d = new Date(date);
  const day = d.getDate();
  d.setDate(1);
  d.setMonth(d.getMonth() + count);
  const maxDay = new Date(d.getFullYear(), d.getMonth()+1, 0).getDate();
  d.setDate(Math.min(day, maxDay));
  return d;
}

function weekStartMonday(date){
  const d = new Date(date);
  const offset = (d.getDay()+6)%7;
  d.setDate(d.getDate()-offset);
  return d;
}

function weekEndSunday(date){
  const d = weekStartMonday(date);
  d.setDate(d.getDate()+6);
  return d;
}

function sameDay(a,b){
  return a.getFullYear()===b.getFullYear() &&
         a.getMonth()===b.getMonth() &&
         a.getDate()===b.getDate();
}

function imagePage(num){
  const pageFile = `page_${String(num).padStart(2,'0')}.png`;
  return `<section class="page static-image-page"><img class="full-page" src="${pageFile}" alt="Page ${num}"></section>`;
}

function coverPage(name, dob, format){
  const dateStr = fmtCoverDate(dob, format);

  return `

    <span class="poem-line">written with love, one day at a time.</span>

  <div class="cover-dynamic-name">${name}</div>
  <div class="cover-dynamic-date">${dateStr}</div>
</section>`;
}

function calendarRows(start, end, birth, monthIndex){
  const begin = weekStartMonday(start);
  const finish = weekEndSunday(end);
  const rows = [];
  let cur = new Date(begin);

  while(cur <= finish){
    const row = [];

    for(let i=0; i<7; i++){
      const c = new Date(cur);
      const inRange = c >= start && c <= end;
      let marker = "";

      if(sameDay(c,birth)){
        marker = "Birth day";
      } else if(sameDay(c,start) && monthIndex > 1){
        marker = `${monthIndex-1} month${monthIndex-1===1?'':'s'} old`;
      }

      row.push({
        num:c.getDate(),
        outside:!inRange,
        marker,
        monthName:(c.getDate()===1)?c.toLocaleDateString('en-AU',{month:'long'}):""
      });

      cur.setDate(cur.getDate()+1);
    }

    rows.push(row);
  }

  return rows.map(row => `
<div class="calendar-row">
  ${row.map(cell => `
    <div class="cell ${cell.outside ? 'outside' : ''}">
      <span class="num">${cell.num}</span>
      ${cell.monthName ? `<span class="month-in-cell">${cell.monthName}</span>` : ''}
      ${cell.marker ? `<span class="marker">${cell.marker}</span>` : ''}
    </div>
  `).join('')}
</div>`).join('');
}

function dynamicCalendarPage(monthIndex, dob){
  const start = addMonthsKeepingDay(dob, monthIndex-1);
  const nextStart = addMonthsKeepingDay(dob, monthIndex);
  const end = new Date(nextStart);
  end.setDate(end.getDate()-1);

  return `
<section class="page calendar-page">
  <div class="calendar-frame"></div>
  <div class="month-title">Month ${monthIndex}</div>

  <div class="calendar-card">
    <h4>Baby Care Tip</h4>

  </div>

  <div class="calendar-grid">

      <div class="weekday">Wed</div>

      <div class="weekday">Fri</div>
      <div class="weekday">Sat</div>
      <div class="weekday">Sun</div>
    </div>

</section>`;
}

function scalePages(){
  const pages = document.querySelectorAll('.page');
  const pageW = 8.5 * 96; // 816px at 96dpi
  const avail = window.innerWidth - 40;
  const scale = Math.min(1, avail / pageW);
  pages.forEach(p => {
    p.style.transform = `scale(${scale})`;
    p.style.marginBottom = `-${(1 - scale) * 11 * 96 * 0.5}px`;
  });
}
 
function renderBook(){
  const name = document.getElementById('babyName').value.trim() || "Baby";
  const dobValue = document.getElementById('babyDob').value;
  const format = document.getElementById('dateFormat').value;

  if(!dobValue){
    alert("Please enter a date of birth.");
    return;
  }

  const dob = new Date(dobValue + "T00:00:00");
  let html = "";

  for(let page=1; page<=50; page++){
    if(page === 1){
      html += coverPage(name, dob, format);
      continue;
    }

    const calendarEntry = Object.entries(dynamicCalendarPages).find(
      ([month, pageNum]) => pageNum === page
    );

    if(calendarEntry){
      html += dynamicCalendarPage(Number(calendarEntry[0]), dob);
      continue;
    }

    html += imagePage(page);
  }

  document.getElementById("book").innerHTML = html;
  scalePages();
}
